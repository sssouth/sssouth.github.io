//=====================================读入信息函数========================================
void Read()
{
    SHOP * head,* p,* end;
    m=0;//记录店面个数
    FILE * fp;
    head=(SHOP * )malloc(sizeof(SHOP));
    end=head;
    end->next=NULL;
    shop=end;//读取文件中的数据,存放在链表中
    fp=fopen("SA.txt","r");
    if(fp==NULL)    //若文件不存在,则创建一个文件
    {
        fp=fopen("SA.txt","w");
        fclose(fp);
        fp=fopen("SA.txt","r");
    }
    p=(SHOP * )malloc(sizeof(SHOP));//fscanf()函数,若读到文件信息则返回读取个数,否则返回-1(EOF)
    while(fscanf(fp,"%d %s %s %s %d %s %d %d %s %s %s\n",&p->num,p->name,p->owner,p->sex,&p->old,p->tele,&p->area,&p->rental,p->kind,p->year,p->isent)!=EOF)
    {
        m++;
        end->next=p;
        end=p;
        p=(SHOP * )malloc(sizeof(SHOP));
    }
    end->next=NULL;
    fclose(fp);
}


//=====================================保存信息函数========================================
void Save()
{
    FILE * fp;
    SHOP * p;
    p=shop->next;//让P指向头节点shop,将信息逐个存入文件中
    fp=fopen("SA.txt","w");
    if(fp==NULL)
    {
        printf("\n\t文件打开失败\n");
        exit(0);
    }

    while(p)
    {
        fprintf(fp,"%d %s %s %s %d %s %d %d %s %s %s\n",p->num,p->name,p->owner,p->sex,p->old,p->tele,p->area,p->rental,p->kind,p->year,p->isent);
        p=p->next;
    }
    fclose(fp);
}


//=====================================判断店面号是否存在函数========================================
int isExisted(int num)
{
    int flag=0;
    SHOP * p=shop->next;
    if(p==NULL)
        return 0;
    while(p)//将链表中的店面号遍历,判断店面号是否存在
    {
        if(p->num==num)
        {
            flag=1;//flag作为店面号存在的标志
            break;
        }
        p=p->next;
    }
    if(flag==0)
        return 0;
    else
        return 1;
}

            break;
        }
        else
            printf("\n\n\t\t\t\t\t\t\t\t输入的用户名或密码错误!\n");

    }
    if(flag==0)

        printf("\n\t\t\t\t\t\t\t\t抱歉,你已不能再继续登陆!\n");
    }

void Register()
{
    FILE * fp;
    int flag=0;
    fp=fopen("User.txt","r");
    if(fp==NULL)
    {
        fp=fopen("User.txt","w");
        fclose(fp);
        fp=fopen("User.txt","r");
    }
    while(true)
    {
        printf("\n\t\t\t\t\t\t\t\t请输入你的用户名:");
        scanf("%s",user);
        while(!feof(fp))
        {
            fscanf(fp,"%s\n",f_user);
            if(strcmp(f_user,user)==0)
            {
                flag=1;
                printf("\n\t\t\t\t\t\t\t\t该用户名已存在,请重新输入!\n");
            }
        }
        fclose(fp);

        if(flag==0)
        {
            fp=fopen("User.txt","a");
            fprintf(fp,"%s\n",user);
            fclose(fp);
            while(true)
            {
                printf("\n\n\t\t\t\t\t\t\t\t请输入你的密码:");
                i=0;
                while((ch=getch())!='\r')
                {
                    putchar(' * ');
                    pass1[i]=ch;
                    i++;
                }
                printf("\n\n\t\t\t\t\t\t\t\t请再次输入你的密码:");
                i=0;
                while((ch=getch())!='\r')
                {
                    putchar(' * ');
                    pass2[i]=ch;
                    i++;
                }
                if(strcmp(pass1,pass2)!=0)
                {
                    printf("\n\n\t\t\t\t\t\t\t密码输入不一致,请重新输入!");
                }
                else

                    fp=fopen("User_Pass.txt","a+");

                    fclose(fp);
                    printf("\n\n\t\t\t\t\t\t\t\t注册成功!\n");
                    break;
                }
            }
        }

        if(flag==0)
            break;     
    }
}
